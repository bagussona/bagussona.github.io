<script>

const table = document.querySelector('table#demo');

const array_of_objects = [
	{
    	"product": "cokelat",
        "stock": 5
    },
    {
    	"product": "strawberry",
        "stock": 8
    },
    {
    	"product": "nanas",
        "stock": 9
    }
]

console.clear()
array_of_objects.map(obj => {
	const rowTable = document.createElement('tr');
    const productTable = document.createElement('td');
    const stockTable = document.createElement('td');
    const test = [productTable, stockTable];
    
    productTable.textContent = obj["product"];
    stockTable.textContent = obj["stock"];
    
    for (const el of test) {
		rowTable.appendChild(el)
    }
    
    // rowTable.appendChild(productTable);
    // rowTable.appendChild(stockTable);
    
    table.appendChild(rowTable);
});
console.log(table);
</script>
